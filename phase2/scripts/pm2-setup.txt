# Các bước đã thực hiện để đảm bảo Persistence (Duy trì ứng dụng)

# 1. Khởi chạy ứng dụng lần đầu
pm2 start main.js --name "product-api"

# 2. Cấu hình startup script (Lệnh này được PM2 sinh ra tự động)
pm2 startup
sudo env PATH=$PATH:/usr/bin /usr/lib/node_modules/pm2/bin/pm2 startup systemd -u ubuntu --hp /home/ubuntu

# 3. Lưu trạng thái để tự động phục hồi sau reboot
pm2 save
